# ── AMADEUS (Voos) ──────────────────────────────────────
# Cadastro GRÁTIS em: https://developers.amadeus.com
# Clique em "Get Started" → crie um app → copie as chaves abaixo
AMADEUS_CLIENT_ID=cole_seu_client_id_aqui
AMADEUS_CLIENT_SECRET=cole_seu_client_secret_aqui

# ── RAPIDAPI (Hotéis via Booking) ───────────────────────
# Cadastro GRÁTIS em: https://rapidapi.com
# Busque "Booking com" → Subscribe (plano gratuito: 100 req/mês)
# Copie o header "X-RapidAPI-Key" do painel
RAPIDAPI_KEY=cole_sua_rapidapi_key_aqui

# ── CONFIGURAÇÕES GERAIS ────────────────────────────────
# "sandbox" usa dados de teste (grátis, sem cobranças)
# "production" usa dados reais (Amadeus cobra por chamada)
AMADEUS_ENV=sandbox
